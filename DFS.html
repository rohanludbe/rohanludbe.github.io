DFS Java Code


import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;

public class DFS{
    public static void main(String[] args) {
      
        int start = 1;
        int V=8;
        boolean[] visited = new boolean[V];
        List<Integer>[] adjList;
        adjList = new List[V];

        for(int i=start;i<V;i++){
            adjList[i] = new ArrayList<>();
        }

        adjList[1].addAll(Arrays.asList(2, 3));
        adjList[2].addAll(Arrays.asList(1, 4,5));
        adjList[3].addAll(Arrays.asList(1, 6,7));
        adjList[4].add(2);
        adjList[5].add(2);
        adjList[6].add(3);
        adjList[7].add(3);

        List<Integer> result = new ArrayList<>(); 
        Soln soln = new Soln(); 
        soln.dfs(start, adjList, visited, result);

        for(int i:result){
            System.out.print(i+" ");
        }
    }
}

class Soln{
    public void dfs(int start,List<Integer>[] adjList,boolean[] visited,List<Integer> result){
        visited[start] = true;
        result.add(start);
        for(int i:adjList[start]){
            if(!visited[i]){
                dfs(i, adjList, visited, result);
            }
        }
    }
}
