class Soln{
    public void dls(int start, int depthLimit, List<Integer>[] adjList, boolean[] visited, List<Integer> result){
        visited[start] = true;
        result.add(start);

        for(int i:adjList[start]){
            if(!visited[i]){
                depthLimit--;
                if(depthLimit >= 0){
                    dls(i, depthLimit, adjList, visited, result);
                }
                depthLimit++;
            }
        }
    }
}
